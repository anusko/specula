029    /**
030     * This Visitor class may be used for a type-based Java Virtual Machine
031     * simulation.
032     * It does not check for correct types on the OperandStack or in the
033     * LocalVariables; nor does it check their sizes are sufficiently big.
034     * Thus, to use this Visitor for bytecode verifying, you have to make sure
035     * externally that the type constraints of the Java Virtual Machine instructions
036     * are satisfied. An InstConstraintVisitor may be used for this.
037     * Anyway, this Visitor does not mandate it. For example, when you
038     * visitIADD(IADD o), then there are two stack slots popped and one
039     * stack slot containing a Type.INT is pushed (where you could also
040     * pop only one slot if you know there are two Type.INT on top of the
041     * stack). Monitor-specific behaviour is not simulated.
042     * 
043     * </P><B>Conventions:</B>
044     *
045     * Type.VOID will never be pushed onto the stack. Type.DOUBLE and Type.LONG
046     * that would normally take up two stack slots (like Double_HIGH and
047     * Double_LOW) are represented by a simple single Type.DOUBLE or Type.LONG
048     * object on the stack here.
049     * If a two-slot type is stored into a local variable, the next variable
050     * is given the type Type.UNKNOWN.
051     * 
052     * WARNING! These classes are a fork of the bcel verifier.
053     *
054     * @see #visitDSTORE(DSTORE o)
055     * @see org.apache.bcel.verifier.structurals.InstConstraintVisitor
056     */
057    public class ExecutionVisitor extends EmptyVisitor implements Visitor{

118            /** Symbolically executes the corresponding Java Virtual Machine instruction. */ 
119            public void visitAALOAD(AALOAD o){
120                    stack().pop();                                                                                                          // pop the index int
121    //System.out.print(stack().peek());
122                    Type t = stack().pop(); // Pop Array type
123                    if (t == Type.NULL){
124                            stack().push(Type.NULL);
125                    }       // Do nothing stackwise --- a NullPointerException is thrown at Run-Time
126                    else{
127                            ArrayType at = (ArrayType) t;   
128                            stack().push(at.getElementType());
129                    }
130            }

324            public void visitDUP_X2(DUP_X2 o){
325                    Type w1 = stack().pop();
326                    Type w2 = stack().pop();
327                    if (w2.getSize() == 2){
328                            stack().push(w1);
329                            stack().push(w2);
330                            stack().push(w1);
331                    }
332                    else{
333                            Type w3 = stack().pop();
334                            stack().push(w1);
335                            stack().push(w3);
336                            stack().push(w2);
337                            stack().push(w1);
338                    }
339            }

374            /** Symbolically executes the corresponding Java Virtual Machine instruction. */ 
375            public void visitDUP2_X2(DUP2_X2 o){
376                    Type t = stack().pop();
377                    if (t.getSize() == 2){
378                            Type u = stack().pop();
379                            if (u.getSize() == 2){
380                                    stack().push(t);
381                                    stack().push(u);
382                                    stack().push(t);
383                            }else{
384                                    Type v = stack().pop();
385                                    stack().push(t);
386                                    stack().push(v);
387                                    stack().push(u);
388                                    stack().push(t);
389                            }
390                    }
391                    else{ //t.getSize() is 1
392                            Type u = stack().pop();
393                            Type v = stack().pop();
394                            if (v.getSize() == 2){
395                                    stack().push(u);
396                                    stack().push(t);
397                                    stack().push(v);
398                                    stack().push(u);
399                                    stack().push(t);
400                            }else{
401                                    Type w = stack().pop();
402                                    stack().push(u);
403                                    stack().push(t);
404                                    stack().push(w);
405                                    stack().push(v);
406                                    stack().push(u);
407                                    stack().push(t);
408                            }
409                    }
410            }

690            /** Symbolically executes the corresponding Java Virtual Machine instruction. */ 
691            public void visitINVOKEINTERFACE(INVOKEINTERFACE o){
692                    stack().pop();  //objectref
693                    for (int i=0; i<o.getArgumentTypes(cpg).length; i++){
694                            stack().pop();
695                    }
696                    // We are sure the invoked method will xRETURN eventually
697                    // We simulate xRETURNs functionality here because we
698                    // don't really "jump into" and simulate the invoked
699                    // method.
700                    if (o.getReturnType(cpg) != Type.VOID){
701                            Type t = o.getReturnType(cpg);
702                            if (    t.equals(Type.BOOLEAN)  ||
