090                    /**
091                     * Returns a clone of the "outgoing" frame situation with respect to the given ExecutionChain.
092                     */     
093                    public Frame getOutFrame(ExecutionPath execChain){
094                            executionPredecessors = execChain;
095    
096                            Frame org;
097    
098                            InstructionContext jsr = lastExecutionJSR();
099    
100                            org = (Frame) outFrames.get(jsr);
101    
102                            if (org == null){
103                                    throw new AssertionViolatedException("outFrame not set! This:\n"+this+"\nExecutionChain: "+getExecutionChain()+"\nOutFrames: '"+outFrames+"'.");
104                            }
105                            return org.getClone();
106                    }
107    
108        public Frame getInFrame() {
109                      Frame org;
110                            
111                            InstructionContext jsr = lastExecutionJSR();
112                            
113                            org = (Frame) inFrames.get(jsr);
114    
115                            if (org == null){
116                                throw new AssertionViolatedException("inFrame not set! This:\n"+this+"\nInFrames: '"+inFrames+"'.");
117          }
118          return org.getClone();
119        }

121                    /**
122                     * "Merges in" (vmspec2, page 146) the "incoming" frame situation;
123                     * executes the instructions symbolically
124                     * and therefore calculates the "outgoing" frame situation.
125                     * Returns: True iff the "incoming" frame situation changed after
126                     * merging with "inFrame".
127                     * The execPreds contain the InstructionContext
128                     * objects executed so far in the correct order. This is just
129                     * one execution path [out of many]. This is needed to correctly
130                     * "merge" in the special case of a RET's successor.
131                     * <B>The InstConstraintVisitor and ExecutionVisitor instances
132                     * must be set up correctly.</B>
133                     * @return true - if and only if the "outgoing" frame situation
134                     * changed from the one before execute()ing.
135                     */
136                    public boolean execute(Frame inFrame, ExecutionPath execPreds, ExecutionVisitor ev){
137    
138                // When merge failed, this is useful to see what are two passes
139                //ExecutionPath oldExecPreds = executionPredecessors;
140    
141                executionPredecessors = execPreds;
142    
143                            //sanity check
144                            if ( (lastExecutionJSR() == null) && (subroutines.subroutineOf(getInstruction()) != subroutines.getTopLevel() ) ){
145                                    //throw new AssertionViolatedException("Huh?! Am I '"+this+"' part of a subroutine or not?");
146                                    return false;
147                            }
148                            if ( (lastExecutionJSR() != null) && (subroutines.subroutineOf(getInstruction()) == subroutines.getTopLevel() ) ){
149                                    //throw new AssertionViolatedException("Huh?! Am I '"+this+"' part of a subroutine or not?");
150                                    return false;
151                            }
152    
153                            Frame inF = (Frame) inFrames.get(lastExecutionJSR());
154                            if (inF == null){// no incoming frame was set, so set it.
155                                    inFrames.put(lastExecutionJSR(), inFrame);
156                                    inF = inFrame;
157                            }
158                            else{// if there was an "old" inFrame
159                                    if (inF.equals(inFrame)){ //shortcut: no need to merge equal frames.
160                                            return false;
161                                    }
162                                    if (! mergeInFrames(inFrame)){
163                                            return false;
164                                    }
165                            }
166                            
167                            // Now we're sure the inFrame has changed!
168                            
169                            // new inFrame is already merged in, see above.         
170                            Frame workingFrame = inF.getClone();
171    
172                            // This executes the Instruction.
173                            // Therefore the workingFrame object is modified.
174    //ExecutionVisitor ev = ExecutionVisitor.getInstance(VerifierFactory.getVerifier(method_gen.getClassName()));
175                            ev.setFrame(workingFrame);
176                            getInstruction().accept(ev);
177                            //getInstruction().accept(ExecutionVisitor.withFrame(workingFrame));
178                            outFrames.put(lastExecutionJSR(), workingFrame);
179    
180                            return true;    // new inFrame was different from old inFrame so merging them
181                                                                                    // yielded a different this.inFrame.
182                    }

185                     * Returns a simple String representation of this InstructionContext.
186                     */
187                    public String toString(){
188                    //TODO: Put information in the brackets, e.g.
189                    //      Is this an ExceptionHandler? Is this a RET? Is this the start of

191                            String ret = getInstruction().toString(false)+"\t[InstructionContext]";
192                            return ret;

195                    /**
196                     * Does the actual merging (vmspec2, page 146).
197                     * Returns true IFF this.inFrame was changed in course of merging with inFrame.
198                     */
199                    private boolean mergeInFrames(Frame inFrame){
200                            // TODO: Can be performance-improved.
201                            Frame inF = inFrames.get(lastExecutionJSR());
202                            OperandStack oldstack = inF.getStack().getClone();
203                            LocalVariables oldlocals = inF.getLocals().getClone();
204                            try{
205                                    inF.getStack().merge(inFrame.getStack());
206                                    inF.getLocals().merge(inFrame.getLocals());
207                            }
208                            catch (StructuralCodeConstraintException sce){
209                                    extendMessageWithFlow(sce);
210                                    throw sce;
211                            }
212                            if (oldstack.equals(inF.getStack()) &&
213                                    oldlocals.equals(inF.getLocals()) ){
214                                    return false;
215                            }
216                            else{
217                                    return true;
218                            }
219                    }

221                    /**
222                     * Returns the control flow execution chain. This is built
223                     * while {@link #execute(Frame, ExecutionPath, ExecutionVisitor) executing} the code represented
224                     * by the surrounding ControlFlowGraph.
225                     */
226                    private String getExecutionChain(){
227                return executionPredecessors.toString()+"\n"+this.toString();
228                    }

232                     * Extends the StructuralCodeConstraintException ("e") object with an at-the-end-extended message.
233                     * This extended message will then reflect the execution flow needed to get to the constraint
234                     * violation that triggered the throwing of the "e" object.
235                     */
236                    private void extendMessageWithFlow(StructuralCodeConstraintException e){
237                            String s = "Execution flow:\n";
238                            e.extendMessage("", s+getExecutionChain());
239                    }

257                    /**
258                     * A utility method that calculates the successors of a given InstructionHandle
259                     * That means, a RET does have successors as defined here.
260                     * A JsrInstruction has its target as its successor
261                     * (opposed to its physical successor) as defined here.
262                     */
263    // TODO: implement caching!
264                    private InstructionHandle[] _getSuccessors(){
265                            final InstructionHandle[] empty = new InstructionHandle[0];
266                            final InstructionHandle[] single = new InstructionHandle[1];
267                            final InstructionHandle[] pair = new InstructionHandle[2];
268                    
269                            Instruction inst = getInstruction().getInstruction();
270                    
271                            if (inst instanceof RET){
272                                    Subroutine s = subroutines.subroutineOf(getInstruction());
273                                    if (s==null){ //return empty; // RET in dead code. "empty" would be the correct answer, but we know something about the surrounding project...
274                                            throw new AssertionViolatedException("Asking for successors of a RET in dead code?!");
275                                    }
276    //TODO: remove
277    throw new AssertionViolatedException("DID YOU REALLY WANT TO ASK FOR RET'S SUCCS?");
278    /*
279                                    InstructionHandle[] jsrs = s.getEnteringJsrInstructions();
280                                    InstructionHandle[] ret = new InstructionHandle[jsrs.length];
281                                    for (int i=0; i<jsrs.length; i++){
282                                            ret[i] = jsrs[i].getNext();
283                                    }
284                                    return ret;
285    */
286                            }
287                    
288                            // Terminates method normally.
289                            if (inst instanceof ReturnInstruction){
290                                    return empty;
291                            }
292                    
293                            // Terminates method abnormally, because JustIce mandates
294                            // subroutines not to be protected by exception handlers.
295                            if (inst instanceof ATHROW){
296                                    return empty;
297                            }
298                    
299                            // See method comment.
300                            if (inst instanceof JsrInstruction){
301                                    single[0] = ((JsrInstruction) inst).getTarget();
302                                    return single;
303                            }
304    
305                            if (inst instanceof GotoInstruction){
306                                    single[0] = ((GotoInstruction) inst).getTarget();
307                                    return single;
308                            }
309    
310                            if (inst instanceof BranchInstruction){
311                                    if (inst instanceof Select){
312                                            // BCEL's getTargets() returns only the non-default targets,
313                                            // thanks to Eli Tilevich for reporting.
314                                            InstructionHandle[] matchTargets = ((Select) inst).getTargets();
315                                            InstructionHandle[] ret = new InstructionHandle[matchTargets.length+1];
316                                            ret[0] = ((Select) inst).getTarget();
317                                            System.arraycopy(matchTargets, 0, ret, 1, matchTargets.length);
318                                            return ret;
319                                    }
320                                    else{
321                                            pair[0] = getInstruction().getNext();
322                                            pair[1] = ((BranchInstruction) inst).getTarget();
323                                            return pair;
324                                    }
325                            }
326    
327                            // default case: Fall through.          
328                            single[0] = getInstruction().getNext();
329                            return single;
330                    }

334            /** The Subroutines object for the method whose control flow is represented by this ControlFlowGraph. */
335            private final Subroutines subroutines;
336    
337            /** The ExceptionHandlers object for the method whose control flow is represented by this ControlFlowGraph. */
338            private final ExceptionHandlers exceptionhandlers;
339    
340            /** All InstructionContext instances of this ControlFlowGraph. */
341            private Map<InstructionHandle, InstructionContext> instructionContexts = new HashMap<InstructionHandle, InstructionContext>(); //keys: InstructionHandle, values: InstructionContextImpl
342    
343            /** 
344             * A Control Flow Graph.
345             */
346            public ControlFlowGraph(MethodGen method_gen){
347                    subroutines = new Subroutines(method_gen);
348                    exceptionhandlers = new ExceptionHandlers(method_gen);
349    
350                    final InstructionHandle[] instructionhandles = method_gen.getInstructionList().getInstructionHandles();
351                    for (final InstructionHandle instructionhandle : instructionhandles){
352                            instructionContexts.put(instructionhandle, new InstructionContextImpl(instructionhandle));
353                    }
354            }
355    
356            /**
357             * Returns the InstructionContext of a given instruction.
358             */
359            public InstructionContext contextOf(InstructionHandle inst){
360                    InstructionContext ic = (InstructionContext) instructionContexts.get(inst);
361                    if (ic == null){
362                            throw new AssertionViolatedException("InstructionContext requested for an InstructionHandle that's not known!");
363                    }
364                    return ic;
365            }
366    
367            /**
368             * Returns the InstructionContext[] of a given InstructionHandle[],
369             * in a naturally ordered manner.
370             */
371            public InstructionContext[] contextsOf(InstructionHandle[] insts){
372                    InstructionContext[] ret = new InstructionContext[insts.length];
373                    for (int i = 0; i < insts.length; i++) {
374                            ret[i] = contextOf(insts[i]);
375                    }
376                    return ret;
377            }
378    
379            /**
380             * Returns an InstructionContext[] with all the InstructionContext instances
381             * for the method whose control flow is represented by this ControlFlowGraph
382             * <B>(NOT ORDERED!)</B>.
383             */
384            public InstructionContext[] getInstructionContexts(){
385                    InstructionContext[] ret = new InstructionContext[instructionContexts.values().size()];
386                    return instructionContexts.values().toArray(ret);
387            }
388    
389            /**
390             * Returns true, if and only if the said instruction is not reachable; that means,
391             * if it is not part of this ControlFlowGraph.
392             */
393            public boolean isDead(InstructionHandle i){
394            return subroutines.subroutineOf(i) == null;
395            }
