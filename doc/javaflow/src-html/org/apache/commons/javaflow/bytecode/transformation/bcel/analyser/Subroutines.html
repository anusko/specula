043    /**
044     * Instances of this class contain information about the subroutines
045     * found in a code array of a method.
046     * This implementation considers the top-level (the instructions
047     * reachable without a JSR or JSR_W starting off from the first
048     * instruction in a code array of a method) being a special subroutine;
049     * see getTopLevel() for that.
050     * Please note that the definition of subroutines in the Java Virtual
051     * Machine Specification, Second Edition is somewhat incomplete.
052     * Therefore, JustIce uses an own, more rigid notion.
053     * Basically, a subroutine is a piece of code that starts at the target
054     * of a JSR of JSR_W instruction and ends at a corresponding RET
055     * instruction. Note also that the control flow of a subroutine
056     * may be complex and non-linear; and that subroutines may be nested.
057     *
058     * WARNING! These classes are a fork of the bcel verifier.
059     *
060     * @see #getTopLevel()
061     */
062    public class Subroutines{
063            /**
064             * This inner class implements the Subroutine interface.
065             */
066            private class SubroutineImpl implements Subroutine{
067                    /**
068                     * UNSET, a symbol for an uninitialized localVariable
069                     * field. This is used for the "top-level" Subroutine;
070                     * i.e. no subroutine.
071                     */
072                    private final int UNSET = -1;
073    
074                    /**
075                     * The Local Variable slot where the first
076                     * instruction of this subroutine (an ASTORE) stores
077                     * the JsrInstruction's ReturnAddress in and
078                     * the RET of this subroutine operates on.
079                     */
080                    private int localVariable = UNSET;
081    
082                    /** The instructions that belong to this subroutine. */
083                    private Set<InstructionHandle> instructions;
084    
085                    /*
086                     * Refer to the Subroutine interface for documentation.
087                     */
088                    public boolean contains(InstructionHandle inst){
089                            return instructions.contains(inst);
090                    }
091    
092                    /**
093                     * The JSR or JSR_W instructions that define this
094                     * subroutine by targeting it.
095                     */
096                    private Set<InstructionHandle> theJSRs = new HashSet<InstructionHandle>();
097    
098                    /**
099                     * The RET instruction that leaves this subroutine.
100                     */
101                    private InstructionHandle theRET;
102    
103                    /**
104                     * Returns a String representation of this object, merely
105                     * for debugging purposes.
106                     * (Internal) Warning: Verbosity on a problematic subroutine may cause
107                     * stack overflow errors due to recursive subSubs() calls.
108                     * Don't use this, then.
109                     */
110                    public String toString(){
111                            String ret = "Subroutine: Local variable is '"+localVariable+"', JSRs are '"+theJSRs+"', RET is '"+theRET+"', Instructions: '"+instructions.toString()+"'.";
112    
113                            ret += " Accessed local variable slots: '";
114                            int[] alv = getAccessedLocalsIndices();
115                            for (int i=0; i<alv.length; i++){
116                                    ret += alv[i]+" ";
117                            }
118                            ret+="'.";
119    
120                            ret += " Recursively (via subsub...routines) accessed local variable slots: '";
121                            alv = getRecursivelyAccessedLocalsIndices();
122                            for (int i=0; i<alv.length; i++){
123                                    ret += alv[i]+" ";
124                            }
125                            ret+="'.";
126    
127                            return ret;
128                    }

130                    /**
131                     * Sets the leaving RET instruction. Must be invoked after all instructions are added.
132                     * Must not be invoked for top-level 'subroutine'.
133                     */
134                    void setLeavingRET() {
135                            if (localVariable == UNSET){
136                                    throw new AssertionViolatedException("setLeavingRET() called for top-level 'subroutine' or forgot to set local variable first.");
137                            }
138                            InstructionHandle ret = null;
139                            for (final InstructionHandle actual : instructions) {
140                                    if (actual.getInstruction() instanceof RET){
141                                            if (ret != null){
142                                                    throw new StructuralCodeConstraintException("Subroutine with more then one RET detected: '"+ret+"' and '"+actual+"'.");
143                                            }
144                                            else{
145                                                    ret = actual;
146                                            }
147                                    }
148                            }
149                            if (ret == null){
150                                    throw new StructuralCodeConstraintException("Subroutine without a RET detected.");
151                            }
152                            if (((RET) ret.getInstruction()).getIndex() != localVariable){
153                                    throw new StructuralCodeConstraintException("Subroutine uses '"+ret+"' which does not match the correct local variable '"+localVariable+"'.");
154                            }
155                            theRET = ret;
156                    }

158                    /*
159                     * Refer to the Subroutine interface for documentation.
160                     */
161                    public InstructionHandle[] getEnteringJsrInstructions(){
162                            if (this == TOPLEVEL) {
163                                    throw new AssertionViolatedException("getLeavingRET() called on top level pseudo-subroutine.");
164                            }
165                            InstructionHandle[] jsrs = new InstructionHandle[theJSRs.size()];
166                            return theJSRs.toArray(jsrs);
167                    }
168    
169                    /**
170                     * Adds a new JSR or JSR_W that has this subroutine as its target.
171                     */
172                    public void addEnteringJsrInstruction(InstructionHandle jsrInst){
173                            if ( (jsrInst == null) || (! (jsrInst.getInstruction() instanceof JsrInstruction))){
174                                    throw new AssertionViolatedException("Expecting JsrInstruction InstructionHandle.");
175                            }
176                            if (localVariable == UNSET){
177                                    throw new AssertionViolatedException("Set the localVariable first!");
178                            }
179                            else{
180                                    // Something is wrong when an ASTORE is targeted that does not operate on the same local variable than the rest of the
181                                    // JsrInstruction-targets and the RET.
182                                    // (We don't know out leader here so we cannot check if we're really targeted!)
183                                    if (localVariable != ((ASTORE) (((JsrInstruction) jsrInst.getInstruction()).getTarget().getInstruction())).getIndex()){
184                                            throw new AssertionViolatedException("Setting a wrong JsrInstruction.");
185                                    }
186                            }
187                            theJSRs.add(jsrInst);
188                    }
189    
190                    /*
191                     * Refer to the Subroutine interface for documentation.
192                     */
193                    public InstructionHandle getLeavingRET(){
194                            if (this == TOPLEVEL) {
195                                    throw new AssertionViolatedException("getLeavingRET() called on top level pseudo-subroutine.");
196                            }
197                            return theRET;
198                    }

201                     * Refer to the Subroutine interface for documentation.
202                     */
203                    public InstructionHandle[] getInstructions(){
204                            InstructionHandle[] ret = new InstructionHandle[instructions.size()];
205                            return (InstructionHandle[]) instructions.toArray(ret);
206                    }

208                    /* Satisfies Subroutine.getRecursivelyAccessedLocalsIndices(). */
209                    public int[] getRecursivelyAccessedLocalsIndices(){
210                            final Set<Integer> s = new HashSet<Integer>();
211                            int[] lvs = getAccessedLocalsIndices();
212                            for (int j = 0; j < lvs.length; j++) {
213                                    s.add(lvs[j]);
214                            }
215                            _getRecursivelyAccessedLocalsIndicesHelper(s, this.subSubs());
216                            
217                            int[] ret = new int[s.size()];
218                            int j = 0;
219                            for (final int v : s) {
220                                    ret[j++] = v;
221                            }
222                            return ret;
223                    }
224    
225                    /**
226                     * A recursive helper method for getRecursivelyAccessedLocalsIndices().
227                     * @see #getRecursivelyAccessedLocalsIndices()
228                     */
229                    private void _getRecursivelyAccessedLocalsIndicesHelper(Set<Integer> s, Subroutine[] subs){
230                            for (final Subroutine sub : subs) {
231                                    for (final int lv : sub.getAccessedLocalsIndices()) {
232                                            s.add(lv);
233                                    }
234                                    
235                                    if(sub.subSubs().length != 0) {
236                                            _getRecursivelyAccessedLocalsIndicesHelper(s, sub.subSubs());
237                                    }
238                            }
239                    }
240    
241                    /*
242                     * Satisfies Subroutine.getAccessedLocalIndices().
243                     */
244                    public int[] getAccessedLocalsIndices(){
245                            //TODO: Implement caching.
246                            final Set<Integer> acc = new HashSet<Integer>();
247                            if (theRET == null && this != TOPLEVEL){
248                                    throw new AssertionViolatedException("This subroutine object must be built up completely before calculating accessed locals.");
249                            }
250                            for (final InstructionHandle ih : instructions) {
251                                    // RET is not a LocalVariableInstruction in the current version of BCEL.
252                                    if (ih.getInstruction() instanceof LocalVariableInstruction || ih.getInstruction() instanceof RET){
253                                            int idx = IndexedInstruction.class.cast(ih.getInstruction()).getIndex();
254                                            acc.add(idx);
255                                            // LONG? DOUBLE?.
256                                            try {
257                                                    // LocalVariableInstruction instances are typed without the need to look into
258                                                    // the constant pool.
259                                                    if (ih.getInstruction() instanceof LocalVariableInstruction) {
260                                                            int s = LocalVariableInstruction.class.cast(ih.getInstruction()).getType(null).getSize();
261                                                            if (s == 2) acc.add(idx + 1);
262                                                    }
263                                            } catch(final RuntimeException re) {
264                                                    throw new AssertionViolatedException("Oops. BCEL did not like NULL as a ConstantPoolGen object.");
265                                            }
266                                    }
267                            }
268    
269                            int[] ret = new int[acc.size()];
270                            int j = 0;
271                            for (final int v : acc) {
272                                    ret[j++] = v;
273                            }
274                            return ret;
275                    }

277                    /*
278                     * Satisfies Subroutine.subSubs().
279                     */
280                    public Subroutine[] subSubs(){
281                            final Set<Subroutine> h = new HashSet<Subroutine>();
282                            for (final InstructionHandle ih : instructions){
283                                    final Instruction inst = ih.getInstruction();
284                                    if (inst instanceof JsrInstruction){
285                                            InstructionHandle targ = ((JsrInstruction) inst).getTarget();
286                                            h.add(getSubroutine(targ));
287                                    }
288                            }
289                            final Subroutine[] ret = new Subroutine[h.size()];
290                            return h.toArray(ret);
291                    }
292    
293                    /*
294                     * Sets the local variable slot the ASTORE that is targeted
295                     * by the JsrInstructions of this subroutine operates on.
296                     * This subroutine's RET operates on that same local variable
297                     * slot, of course.
298                     */
299                    void setLocalVariable(int i) {
300                            if (localVariable != UNSET){
301                                    throw new AssertionViolatedException("localVariable set twice.");
302                            }
303                            else{
304                                    localVariable = i;
305                            }
306                    }

319            /**
320             * The Hashtable containing the subroutines found.
321             * Key: InstructionHandle of the leader of the subroutine.
322             * Elements: SubroutineImpl objects.
323             */
324            private Map<InstructionHandle, Subroutine> subroutines = new HashMap<InstructionHandle, Subroutine>();
325    
326            /**
327             * This is referring to a special subroutine, namely the
328             * top level. This is not really a subroutine but we use
329             * it to distinguish between top level instructions and
330             * unreachable instructions.
331             */
332            public final Subroutine TOPLEVEL;
333    
334            /**
335             * Constructor.
336             * @param mg A MethodGen object representing method to
337             * create the Subroutine objects of.
338             */
339            public Subroutines(MethodGen mg){
340    
341                    InstructionHandle[] all = mg.getInstructionList().getInstructionHandles();
342                    CodeExceptionGen[] handlers = mg.getExceptionHandlers();
343    
344                    // Define our "Toplevel" fake subroutine.
345                    TOPLEVEL = new SubroutineImpl();
346    
347                    // Calculate "real" subroutines.
348                    Set<InstructionHandle> sub_leaders = new HashSet<InstructionHandle>(); // Elements: InstructionHandle
349                    for (final InstructionHandle ih : all) {
350                            Instruction inst = ih.getInstruction();
351                            if (inst instanceof JsrInstruction){
352                                    sub_leaders.add(((JsrInstruction) inst).getTarget());
353                            }
354                    }
355    
356                    // Build up the database.
357                    for (final InstructionHandle astore : sub_leaders){
358                            final SubroutineImpl sr = new SubroutineImpl();
359                            sr.setLocalVariable( ASTORE.class.cast(astore.getInstruction()).getIndex() );
360                            subroutines.put(astore, sr);
361                    }
362    
363                    // Fake it a bit. We want a virtual "TopLevel" subroutine.
364                    subroutines.put(all[0], TOPLEVEL);
365                    sub_leaders.add(all[0]);
366    
367                    // Tell the subroutines about their JsrInstructions.
368                    // Note that there cannot be a JSR targeting the top-level
369                    // since "Jsr 0" is disallowed in Pass 3a.
370                    // Instructions shared by a subroutine and the toplevel are
371                    // disallowed and checked below, after the BFS.
372                    for (final InstructionHandle ih : all){
373                            Instruction inst = ih.getInstruction();
374                            if (inst instanceof JsrInstruction){
375                                    InstructionHandle leader = ((JsrInstruction) inst).getTarget();
376                                    ((SubroutineImpl) getSubroutine(leader)).addEnteringJsrInstruction(ih);
377                            }
378                    }
379    
380                    // Now do a closure computation from every subroutine leader to find all the
381                    // instructions that belong to a subroutine.
382                    final Set<InstructionHandle> instructions_assigned = new HashSet<InstructionHandle>(); // we don't want to assign an instruction to two or more Subroutine objects.
383                    for (final InstructionHandle leader : sub_leaders){
384                // set of InstructionHandles reachable from the top
385                final Set<InstructionHandle> closure = new HashSet<InstructionHandle>();
386    
387                // Init Queue. Start with the entry point of closure.
388                            List<InstructionHandle> Q = new ArrayList<InstructionHandle>();
389                            Q.add(leader);
390    
391                while (!Q.isEmpty()) {
392                    while (!Q.isEmpty()) {
393                        InstructionHandle u = (InstructionHandle) Q.remove(Q.size() - 1);
394                        if (closure.add(u)) {
395                            for (final InstructionHandle successor : getSuccessors(u)) {
396                                Q.add(successor);
397                            }
398                        }
399                    }
400                    // check if exception handlers are reachable from the closure computed thus far.
401                    // conceptually, an exception handler is a successor of each instruction inside
402                    // the protected region, but the computation works faster if we don't implement it
403                    // in a straight-forward way. Instead, we occasionally check if the first instruction
404                    // of the protected region is in the closure, and if so, we know that the exception
405                    // handler should also be in the closure.
406                    //
407                    // "one instruction must be always in at-most one subroutine" JVM requirement
408                    // guarantees that the entire protected region always belongs to the same closure,
409                    // so just checking the start instruction is suffice.
410                    for(final CodeExceptionGen handler : handlers) {
411                        if (closure.contains(handler.getStartPC())) {
412                            InstructionHandle handlerPC = handler.getHandlerPC();
413                            if (!closure.contains(handlerPC)) {
414                                Q.add(handlerPC);
415                            }
416                        }
417                    }
418                }
419                // DFS ended above.
420                SubroutineImpl.class.cast(leader == all[0] ? getTopLevel() : getSubroutine(leader)).setInstructions(closure);
421    
422                for (final InstructionHandle h : closure) {
423                    if (!instructions_assigned.add(h)) {
424                        throw new StructuralCodeConstraintException("Instruction '" + h + "' is part of more than one subroutine (or of the top level and a subroutine).");
425                    }
426                }
427    
428                if (leader != all[0]){// If we don't deal with the top-level 'subroutine'
429                                    ((SubroutineImpl) getSubroutine(leader)).setLeavingRET();
430                            }
431                    }
432    
433                    // Now make sure no subroutine is calling a subroutine
434                    // that uses the same local variable for the RET as themselves
435                    // (recursively).
436                    // This includes that subroutines may not call themselves
437                    // recursively, even not through intermediate calls to other
438                    // subroutines.
439                    noRecursiveCalls(getTopLevel(), new HashSet<Integer>());
440    
441            }
442    
443            /**
444             * This (recursive) utility method makes sure that
445             * no subroutine is calling a subroutine
446             * that uses the same local variable for the RET as themselves
447             * (recursively).
448             * This includes that subroutines may not call themselves
449             * recursively, even not through intermediate calls to other
450             * subroutines.
451             *
452             * @throws StructuralCodeConstraintException if the above constraint is not satisfied.
453             */
454            private void noRecursiveCalls(final Subroutine sub, final Set<Integer> set){
455                    for (final Subroutine subSub : sub.subSubs()) {
456                            int index = ((RET) (subSub.getLeavingRET().getInstruction())).getIndex();
457                            if (!set.add(index)) {
458                                    // Don't use toString() here because of possibly infinite recursive subSubs() calls then.
459                                    final SubroutineImpl si = (SubroutineImpl)subSub;
460                                    throw new StructuralCodeConstraintException("Subroutine with local variable '" + si.localVariable + "', JSRs '" + si.theJSRs+"', RET '" + si.theRET + "' is called by a subroutine which uses the same local variable index as itself; maybe even a recursive call? JustIce's clean definition of a subroutine forbids both.");
461                            }
462    
463                            noRecursiveCalls(subSub, set);
464    
465                            set.remove(index);
466                    }
467            }

469            /**
470             * Returns the Subroutine object associated with the given
471             * leader (that is, the first instruction of the subroutine).
472             * You must not use this to get the top-level instructions
473             * modeled as a Subroutine object.
474             *
475             * @see #getTopLevel()
476             */
477            public Subroutine getSubroutine(InstructionHandle leader){
478                    Subroutine ret = (Subroutine) subroutines.get(leader);
479    
480                    if (ret == null){
481                            throw new AssertionViolatedException("Subroutine requested for an InstructionHandle that is not a leader of a subroutine.");
482                    }
483    
484                    if (ret == TOPLEVEL){
485                            throw new AssertionViolatedException("TOPLEVEL special subroutine requested; use getTopLevel().");
486                    }
487    
488                    return ret;
489            }

491            /**
492             * Returns the subroutine object associated with the
493             * given instruction. This is a costly operation, you
494             * should consider using getSubroutine(InstructionHandle).
495             * Returns 'null' if the given InstructionHandle lies
496             * in so-called 'dead code', i.e. code that can never
497             * be executed.
498             *
499             * @see #getSubroutine(InstructionHandle)
500             * @see #getTopLevel()
501             */
502            public Subroutine subroutineOf(InstructionHandle any){
503                    for (final Subroutine s : subroutines.values()) {
504                            if (s.contains(any)) 
505                                    return s;
506                    }
507    System.err.println("DEBUG: Please verify '" + any + "' lies in dead code.");
508                    return null;
509                    //throw new AssertionViolatedException("No subroutine for InstructionHandle found (DEAD CODE?).");
510            }

513             * For easy handling, the piece of code that is <B>not</B> a
514             * subroutine, the top-level, is also modeled as a Subroutine
515             * object.
516             * It is a special Subroutine object where <B>you must not invoke
517             * getEnteringJsrInstructions() or getLeavingRET()</B>.
518             *
519             * @see Subroutine#getEnteringJsrInstructions()
520             * @see Subroutine#getLeavingRET()
521             */
522            public Subroutine getTopLevel(){
523                    return TOPLEVEL;
524            }

526        static final InstructionHandle[] empty = new InstructionHandle[0];
527    
528        /**
529             * A utility method that calculates the successors of a given InstructionHandle
530             * <B>in the same subroutine</B>. That means, a RET does not have any successors
531             * as defined here. A JsrInstruction has its physical successor as its successor
532             * (opposed to its target) as defined here.
533             */
534            private static InstructionHandle[] getSuccessors(InstructionHandle instruction){
535                    Instruction inst = instruction.getInstruction();
536    
537            if (inst instanceof RET){
538                            return empty;
539                    }
540    
541                    // Terminates method normally.
542                    if (inst instanceof ReturnInstruction){
543                            return empty;
544                    }
545    
546                    // Terminates method abnormally, because JustIce mandates
547                    // subroutines not to be protected by exception handlers.
548                    if (inst instanceof ATHROW){
549                            return empty;
550                    }
551    
552            final InstructionHandle[] single = new InstructionHandle[1];
553    
554            // See method comment.
555                    if (inst instanceof JsrInstruction){
556                            single[0] = instruction.getNext();
557                            return single;
558                    }
559    
560                    if (inst instanceof GotoInstruction){
561                            single[0] = ((GotoInstruction) inst).getTarget();
562                            return single;
563                    }
564    
565                    if (inst instanceof BranchInstruction){
566                            if (inst instanceof Select){
567                                    // BCEL's getTargets() returns only the non-default targets,
568                                    // thanks to Eli Tilevich for reporting.
569                                    InstructionHandle[] matchTargets = ((Select) inst).getTargets();
570                                    InstructionHandle[] ret = new InstructionHandle[matchTargets.length+1];
571                                    ret[0] = ((Select) inst).getTarget();
572                                    System.arraycopy(matchTargets, 0, ret, 1, matchTargets.length);
573                                    return ret;
574                            }
575                            else{
576                    final InstructionHandle[] pair = new InstructionHandle[2];
577                                    pair[0] = instruction.getNext();
578                                    pair[1] = ((BranchInstruction) inst).getTarget();
579                                    return pair;
580                            }
581                    }
582    
583                    // default case: Fall through.
584                    single[0] = instruction.getNext();
585                    return single;
586            }
