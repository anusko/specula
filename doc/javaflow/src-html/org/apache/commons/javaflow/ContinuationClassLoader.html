017    package org.apache.commons.javaflow;
018    
019    import org.apache.commons.javaflow.bytecode.transformation.ResourceTransformer;
020    import org.apache.commons.javaflow.bytecode.transformation.asm.AsmClassTransformer;
021    import org.apache.commons.logging.Log;
022    import org.apache.commons.logging.LogFactory;
023    
024    import java.io.ByteArrayOutputStream;
025    import java.io.IOException;
026    import java.io.InputStream;
027    import java.net.URL;
028    import java.net.URLClassLoader;
029    import java.security.AccessControlContext;
030    import java.security.AccessController;
031    import java.security.PrivilegedAction;
032    import java.security.ProtectionDomain;
033    import java.util.ArrayList;
034    import java.util.List;

036    /**
037     * {@link URLClassLoader} with bytecode instrumentation for javaflow.
038     *
039     * <p>
040     * This class loader is useful where the application can set up multiple
041     * class loaders (such as in a container environment,
042     * <a href="http://classworlds.codehaus.org/">ClassWorlds</a>, or
043     * <a href="http://forehead.werken.com/">Forehead</a>) and when you can
044     * isolate the continuation-enabled portion of your application into a separate
045     * jar file.
046     */
047    public final class ContinuationClassLoader extends URLClassLoader {
048    
049        private final static Log log = LogFactory.getLog(ContinuationClassLoader.class);
050    
051        private final ResourceTransformer transformer;
052    
053        /**
054         * Indicates whether the parent class loader should be
055         * consulted before trying to load with this class loader.
056         */
057        private boolean parentFirst = true;
058    
059        /**
060         * These are the package roots that are to be loaded by the parent class
061         * loader regardless of whether the parent class loader is being searched
062         * first or not.
063         */
064        private List<String> systemPackages = new ArrayList<String>();
065    
066        /**
067         * These are the package roots that are to be loaded by this class loader
068         * regardless of whether the parent class loader is being searched first
069         * or not.
070         */
071        private List<String> loaderPackages = new ArrayList<String>();
072    
073        /**
074         * Whether or not this classloader will ignore the base
075         * classloader if it can't find a class.
076         *
077         * @see #setIsolated(boolean)
078         */
079        private boolean ignoreBase = false;
080    
081        /* The context to be used when loading classes and resources */
082        private final AccessControlContext acc;
083    
084        private static final int BUFFER_SIZE = 4096;
085    
086        /**
087         * Creates a classloader by using the classpath given.
088         *
089         * @param urls
090         *      The URLs from which to load classes and resources
091         * @param parent
092         *      The parent classloader to which unsatisfied loading
093         *      attempts are delegated. May be <code>null</code>,
094         *      in which case the {@link ClassLoader#getSystemClassLoader() system classloader}
095         *      is used as the parent.
096         * @param transformer
097         *      This transformer is used to perform the byte-code enhancement.
098         *      May not be null.
099         */
100        public ContinuationClassLoader(URL[] urls, ClassLoader parent, ResourceTransformer transformer) {
101            super(urls,fixNullParent(parent));
102            if(transformer==null)
103                throw new IllegalArgumentException();
104            this.transformer = transformer;
105            acc = AccessController.getContext();
106        }
107    
108        public ContinuationClassLoader(URL[] urls, ClassLoader parent) {
109            this(urls,parent,new AsmClassTransformer());
110        }
111    
112        private static ClassLoader fixNullParent(ClassLoader classLoader) {
113            if(classLoader!=null) {
114                return classLoader;
115            } else {
116                return getSystemClassLoader();
117            }
118        }
119    
120        /**
121         * Control whether class lookup is delegated to the parent loader first
122         * or after this loader. Use with extreme caution. Setting this to
123         * false violates the class loader hierarchy and can lead to Linkage errors
124         *
125         * @param parentFirst if true, delegate initial class search to the parent
126         *                    classloader.
127         */
128        public void setParentFirst(boolean parentFirst) {
129            this.parentFirst = parentFirst;
130        }
131    
132        /**
133         * Sets whether this classloader should run in isolated mode. In
134         * isolated mode, classes not found on the given classpath will
135         * not be referred to the parent class loader but will cause a
136         * ClassNotFoundException.
137         *
138         * @param isolated Whether or not this classloader should run in
139         *                 isolated mode.
140         */
141        public void setIsolated(boolean isolated) {
142            ignoreBase = isolated;
143        }
144    
145        /**
146         * Adds a package root to the list of packages which must be loaded on the
147         * parent loader.
148         *
149         * All subpackages are also included.
150         *
151         * @param packageRoot The root of all packages to be included.
152         *                    Should not be <code>null</code>.
153         */
154        public synchronized void addSystemPackageRoot(String packageRoot) {
155            systemPackages.add(appendDot(packageRoot));
156        }
157    
158        /**
159         * Adds a package root to the list of packages which must be loaded using
160         * this loader.
161         *
162         * All subpackages are also included.
163         *
164         * @param packageRoot The root of all packages to be included.
165         *                    Should not be <code>null</code>.
166         */
167        public synchronized void addLoaderPackageRoot(String packageRoot) {
168            loaderPackages.add(appendDot(packageRoot));
169        }
170    
171        private String appendDot(String str) {
172            if(str.endsWith("."))
173                str += '.';
174            return str;
175        }
176    
177        /**
178         * Loads a class through this class loader even if that class is available
179         * on the parent classpath.
180         *
181         * This ensures that any classes which are loaded by the returned class
182         * will use this classloader.
183         *
184         * @param classname The name of the class to be loaded.
185         *                  Must not be <code>null</code>.
186         *
187         * @return the required Class object
188         *
189         * @exception ClassNotFoundException if the requested class does not exist
190         *                                   on this loader's classpath.
191         */
192        public Class<?> forceLoadClass(String classname)
193             throws ClassNotFoundException {
194            log.debug("force loading " + classname);
195    
196            Class<?> theClass = findLoadedClass(classname);
197    
198            if (theClass == null) {
199                theClass = findClass(classname);
200            }
201    
202            return theClass;
203        }
204    
205        /**
206         * Tests whether or not the parent classloader should be checked for
207         * a resource before this one. If the resource matches both the
208         * "use parent classloader first" and the "use this classloader first"
209         * lists, the latter takes priority.
210         *
211         * @param resourceName The name of the resource to check.
212         *                     Must not be <code>null</code>.
213         *
214         * @return whether or not the parent classloader should be checked for a
215         *         resource before this one is.
216         */
217        private synchronized boolean isParentFirst(String resourceName) {
218            // default to the global setting and then see
219            // if this class belongs to a package which has been
220            // designated to use a specific loader first
221            // (this one or the parent one)
222    
223            // XXX - shouldn't this always return false in isolated mode?
224    
225            boolean useParentFirst = parentFirst;
226    
227            for (final String packageName : systemPackages) {
228                if (resourceName.startsWith(packageName)) {
229                    useParentFirst = true;
230                    break;
231                }
232            }
233    
234            for (final String packageName : loaderPackages) {
235                if (resourceName.startsWith(packageName)) {
236                    useParentFirst = false;
237                    break;
238                }
239            }
240    
241            return useParentFirst;
242        }
243    
244        /**
245         * Loads a class with this class loader.
246         *
247         * This class attempts to load the class in an order determined by whether
248         * or not the class matches the system/loader package lists, with the
249         * loader package list taking priority. If the classloader is in isolated
250         * mode, failure to load the class in this loader will result in a
251         * ClassNotFoundException.
252         *
253         * @param classname The name of the class to be loaded.
254         *                  Must not be <code>null</code>.
255         * @param resolve <code>true</code> if all classes upon which this class
256         *                depends are to be loaded.
257         *
258         * @return the required Class object
259         *
260         * @exception ClassNotFoundException if the requested class does not exist
261         * on the system classpath (when not in isolated mode) or this loader's
262         * classpath.
263         */
264        protected synchronized Class<?> loadClass(String classname, boolean resolve)
265             throws ClassNotFoundException {
266            // 'sync' is needed - otherwise 2 threads can load the same class
267            // twice, resulting in LinkageError: duplicated class definition.
268            // findLoadedClass avoids that, but without sync it won't work.
269    
270            Class<?> theClass = findLoadedClass(classname);
271            if (theClass != null) {
272                return theClass;
273            }
274    
275            if (isParentFirst(classname)) {
276                try {
277                    theClass = getParent().loadClass(classname);
278                    log.debug("Class " + classname + " loaded from parent loader "
279                        + "(parentFirst)");
280                } catch (ClassNotFoundException cnfe) {
281                    theClass = findClass(classname);
282                    log.debug("Class " + classname + " loaded from ant loader "
283                        + "(parentFirst)");
284                }
285            } else {
286                try {
287                    theClass = findClass(classname);
288                    log.debug("Class " + classname + " loaded from ant loader");
289                } catch (ClassNotFoundException cnfe) {
290                    if (ignoreBase) {
291                        throw cnfe;
292                    }
293                    theClass = getParent().loadClass(classname);
294                    log.debug("Class " + classname + " loaded from parent loader");
295                }
296            }
297    
298            if (resolve) {
299                resolveClass(theClass);
300            }
301    
302            return theClass;
303        }
304    
305        /**
306         * Define a class given its bytes
307         *
308         * @param classData the bytecode data for the class
309         * @param classname the name of the class
310         *
311         * @return the Class instance created from the given data
312         */
313        protected Class<?> defineClassFromData(final byte[] classData, final String classname) {
314            return AccessController.doPrivileged(new PrivilegedAction<Class<?>>() {
315                public Class<?> run() {
316                    // define a package if necessary.
317                    int i = classname.lastIndexOf('.');
318                    if (i>0) {
319                        final String packageName = classname.substring(0,i);
320                        final Package pkg = getPackage(packageName);
321                        if (pkg == null) {
322                            definePackage(packageName,null,null,null,null,null,null,null);
323                        }
324                    }
325    
326                    final byte[] newData = transformer.transform(classData);
327                    final ProtectionDomain domain = this.getClass().getProtectionDomain();
328                    return defineClass(classname,newData,0,newData.length,domain);
329                }
330            }, acc);
331        }
332    
333        /**
334         * Reads a class definition from a stream.
335         *
336         * @param stream The stream from which the class is to be read.
337         *               Must not be <code>null</code>.
338         * @param classname The name of the class in the stream.
339         *                  Must not be <code>null</code>.
340         *
341         * @return the Class object read from the stream.
342         *
343         * @exception IOException if there is a problem reading the class from the
344         * stream.
345         * @exception SecurityException if there is a security problem while
346         * reading the class from the stream.
347         */
348        private Class<?> getClassFromStream(InputStream stream, String classname) throws IOException, SecurityException {
349    
350            final ByteArrayOutputStream baos = new ByteArrayOutputStream();
351            try {
352                    
353                    int bytesRead;
354                    byte[] buffer = new byte[BUFFER_SIZE];
355            
356                    while ((bytesRead = stream.read(buffer, 0, BUFFER_SIZE)) != -1) {
357                        baos.write(buffer, 0, bytesRead);
358                    }
359            
360                    byte[] classData = baos.toByteArray();
361                    return defineClassFromData(classData, classname);
362            
363            } finally {
364                    baos.close();
365            }
366        }
367    
368        /**
369         * Searches for and load a class on the classpath of this class loader.
370         *
371         * @param name The name of the class to be loaded. Must not be
372         *             <code>null</code>.
373         *
374         * @return the required Class object
375         *
376         * @exception ClassNotFoundException if the requested class does not exist
377         *                                   on this loader's classpath.
378         */
379        public Class<?> findClass(final String name) throws ClassNotFoundException {
380            log.debug("Finding class " + name);
381    
382            // locate the class file
383            String classFileName = name.replace('.', '/') + ".class";
384    
385            InputStream stream = getResourceAsStream(classFileName);
386            if(stream==null)
387                throw new ClassNotFoundException(name);
388    
389            try {
390                return getClassFromStream(stream, name);
391            } catch (IOException e) {
392                throw new ClassNotFoundException(name,e);
393            } finally {
394                try {
395                    stream.close();
396                } catch (IOException e) {
397                    // ignore
398                }
399            }
400        }
401    
402        /**
403         * Finds the resource with the given name. A resource is
404         * some data (images, audio, text, etc) that can be accessed by class
405         * code in a way that is independent of the location of the code.
406         *
407         * @param name The name of the resource for which a stream is required.
408         *             Must not be <code>null</code>.
409         * @return a URL for reading the resource, or <code>null</code> if the
410         *         resource could not be found or the caller doesn't have
411         *         adequate privileges to get the resource.
412         */
413        public synchronized URL getResource(String name) {
414            // we need to search the components of the path to see if
415            // we can find the class we want.
416            if (isParentFirst(name)) {
417                return super.getResource(name);
418            }
419    
420            // try this class loader first, then parent
421            URL url = findResource(name);
422            if(url==null) {
423                url = getParent().getResource(name);
424            }
425            return url;
426        }
